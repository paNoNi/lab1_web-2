import numpy as np



def cost_function(features, results, theta):

	rows = len(results)

 	mse = np.sum((np.matmul(theta.transpose(), features[0].transpose()) -
	results. transpose() ) ** 2)

 	return mse / (2 * rows)



def gradient_descent(features, results, theta, alpha, iterations):

 	rows = len(results)

 	cost_func = []

 	for i in range(iterations):

 		res_hyp = np.matmul(theta.transpose(), features.transpose())

 		gradient = np.matmul(res_hyp - results.transpose(), features)

 		theta = (theta.transpose() - alpha * gradient / rows).transpose()

 		cost_func.append((cost_function(features, results, theta)))


 	return [theta, cost_func]
